Invoker Commands API - Custom command


						<img src="https://picsum.photos/id/237/1440/810"
							id="myimage" alt="Random Lorem Picsum image">
						<button
							commandfor="myimage" command="--change"
						>Change image</button>
					

						document.querySelector('#myimage').addEventListener('command',
							(evt) => {
								if (evt.command === '--change') {
									const id = [237, 353, 182][Math.round(Math.random() * 4)];
									evt.target.src = `https://picsum.photos/id/${id}/1440/810`;
								}
							});
