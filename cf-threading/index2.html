Array parallel processing


					array = ["item1", "item2", "item3", "item4", "item5"];

					start = getTickCount();
					arrayEach(
						array=array,
						closure=(item) => {
							sleep(1000);
							writeOutput("#item# #now()#
");
						},
						parallel=true,
						maxThreads=3
					);
					dump("Processing completed in #getTickCount() - start# ms");

Shared data


					request.counter = 0;

					start = getTickCount();
					for (i = 1; i <= 20; i++) {
						thread {
							sleep(1000);
							request.counter++;
						}
					}
					threadJoin();
					dump(var=request.counter, label="All threads completed");
					dump("Total execution time: #getTickCount()-start#ms");

Shared data


					request.counter = 0;

					start = getTickCount();
					for (i = 1; i <= 20; i++) {
						thread {
							sleep(1000);
							lock scope="request" {
								request.counter++;
							}
						}
					}
					threadJoin();
					dump(var=request.counter, label="All threads completed");
					dump("Total execution time: #getTickCount()-start#ms");
